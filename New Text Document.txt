    PORTC EQU 08H

MOV AX, DATA
MOV DS, AX
ORG 0000H
START: 	MOV DX, PORT_CON
	 MOV AL, 10000000B
	 OUT DX, AL
	 MOV BL,10H
	 LEA SI,DIG
NEXT:
	    MOV DX,PORTB
	    MOV AL,[SI]
	    OUT DX,AL
	    MOV CX,8FFFH
DELAY:NOP
	    NOP
	    NOP
	    LOOP DELAY
	    MOV AL, 00H
	    OUT DX,AL
	    MOV CX,08FFH
DELAY1:NOP
	    NOP
	    NOP
	    LOOP DELAY1
	    INC SI
	    DEC BL
	    JNZ NEXT

CODE ENDS
      END START
